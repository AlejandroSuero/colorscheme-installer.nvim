INSTALLATION                         *colorscheme-installer.nvim-installation*

This plugin supports lazy.nvim <https://github.com/folke/lazy.nvim>
installation.

- Default installation:

>lua
    return {
        "AlejandroSuero/colorscheme-installer.nvim",
        config = function()
            require("colorscheme-installer").setup()
        end,
    }
<

- Customizable installation:

>lua
    return {
        "AlejandroSuero/colorscheme-installer.nvim",
        config = function()
            require("colorscheme-installer.nvim").setup({
                -- your configuration goes here
            })
        end,
    }
<


  [!note] See default configuration.

DEFAULT CONFIGURATION       *colorscheme-installer.nvim-default-configuration*

>lua
    local opts = {
        -- path where the colorschemes will be configured
        colorschemes_path = os.getenv("HOME") .. "/.config/nvim/lua/colorschemes"
    }
<
